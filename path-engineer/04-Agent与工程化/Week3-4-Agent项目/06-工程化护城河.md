## 1. 限流（Resilience4j）

### 1.1 配置

```yaml
resilience4j:
  ratelimiter:
    instances:
      llm-call:
        limit-for-period: 100         # 每分钟 100 次
        limit-refresh-period: 60s
        timeout-duration: 0
      per-repo:
        limit-for-period: 20
        limit-refresh-period: 60s
        timeout-duration: 0
```

### 1.2 用注解

```java
@Service
public class RateLimitedReviewer {

    private final ReviewAgent reviewer;
    private final RateLimiterRegistry registry;

    @RateLimiter(name = "llm-call")
    public List<ReviewComment> reviewWithLimit(String f, String d) {
        return reviewer.review(f, d);
    }

    /** 按仓库限流 */
    public List<ReviewComment> reviewPerRepo(String repo, String f, String d) {
        RateLimiter limiter = registry.rateLimiter("repo:" + repo);
        return limiter.executeSupplier(() -> reviewer.review(f, d));
    }
}
```

## 2. 缓存（Redis + 内容哈希）

```java
@Service
@RequiredArgsConstructor
public class CachedReviewer {

    private final ReviewAgent agent;
    private final StringRedisTemplate redis;
    private final ObjectMapper om = new ObjectMapper();

    public List<ReviewComment> review(String filename, String diff) {
        String key = "review:" + sha256(filename + ":" + diff);
        String cached = redis.opsForValue().get(key);
        if (cached != null) {
            log.info("✅ Cache hit: {}", key);
            return parse(cached);
        }
        var result = agent.review(filename, diff);
        redis.opsForValue().set(key, om.writeValueAsString(result), Duration.ofDays(7));
        return result;
    }
}
```

> 🎯 **简历数字**：「LLM 调用缓存命中率 ~ 30%（同一 PR 多次 push 几乎不重审），节省成本 X%」。

## 3. 降级（Circuit Breaker）

```yaml
resilience4j:
  circuitbreaker:
    instances:
      llm-call:
        failure-rate-threshold: 50    # 失败率 >50%
        wait-duration-in-open-state: 30s
        sliding-window-size: 10
```

```java
@CircuitBreaker(name = "llm-call", fallbackMethod = "fallbackReview")
public List<ReviewComment> reviewWithBreaker(String f, String d) {
    return agent.review(f, d);
}

public List<ReviewComment> fallbackReview(String f, String d, Throwable t) {
    log.warn("LLM unavailable, fallback empty review: {}", t.getMessage());
    // 可以兜底为"留一条 summary 评论"
    return List.of();
}
```

### 4.2 自定义指标

```java
@Component
@RequiredArgsConstructor
public class ReviewMetrics {

    private final MeterRegistry meter;

    public void recordReview(String repo, int comments, long elapsedMs) {
        meter.counter("reviewer.pr.processed", "repo", repo).increment();
        meter.summary("reviewer.comments.count", "repo", repo).record(comments);
        meter.timer("reviewer.elapsed", "repo", repo)
             .record(elapsedMs, TimeUnit.MILLISECONDS);
    }

    public void recordLlmCall(String agent, int promptTokens, int outTokens, BigDecimal cost) {
        meter.counter("llm.call.count", "agent", agent).increment();
        meter.summary("llm.tokens", "agent", agent, "type", "prompt").record(promptTokens);
        meter.summary("llm.tokens", "agent", agent, "type", "output").record(outTokens);
        meter.summary("llm.cost", "agent", agent).record(cost.doubleValue());
    }
}
```

## 5. 审计日志

```java
@Aspect
@Component
@Slf4j
public class ReviewAuditAspect {

    @AfterReturning(pointcut = "execution(* *.review(..))", returning = "result")
    public void log(JoinPoint pt, Object result) {
        log.info("audit | method={} | args={} | result_size={}",
            pt.getSignature().getName(),
            Arrays.toString(pt.getArgs()),
            result instanceof Collection<?> c ? c.size() : 1);
    }
}
```

更严的话，落到独立 audit_log 表 / Elasticsearch。
