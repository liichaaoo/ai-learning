## 1. 数据库 Schema（ER 速记图）⭐

```
user (id, username, password_hash, tenant_id, role, created_at)
   │
   ├─► document (id, user_id, tenant_id, filename, status, total_chunks, created_at, sha256)
   │      │
   │      └─► chunk (id, document_id, content, vector_id, chunk_seq, page_num)
   │             ↑
   │             └── vector_id 对应 Milvus 的 entity id
   │
   ├─► chat_session (id, user_id, tenant_id, title, created_at)
   │      │
   │      └─► chat_message (id, session_id, role, content, citations_json, tokens, created_at)
   │
   └─► usage_record (id, user_id, model, prompt_tokens, completion_tokens, cost_yuan, created_at)
```

> 💡 关键设计点：**Milvus 只存向量 + content + metadata；MySQL 是"事实之源"**——chunk 删除时同步删两边。

## 2. Flyway 脚本（节选）

```sql
-- V1__init.sql
CREATE TABLE user (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(64) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    tenant_id VARCHAR(64) NOT NULL,
    role VARCHAR(16) DEFAULT 'USER',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_tenant (tenant_id)
);

CREATE TABLE document (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    tenant_id VARCHAR(64) NOT NULL,
    filename VARCHAR(255) NOT NULL,
    sha256 VARCHAR(64) NOT NULL,
    status VARCHAR(16) DEFAULT 'PENDING',  -- PENDING/PROCESSING/READY/FAILED
    total_chunks INT DEFAULT 0,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_tenant_sha (tenant_id, sha256),
    INDEX idx_user (user_id)
);

CREATE TABLE chunk (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    document_id BIGINT NOT NULL,
    vector_id VARCHAR(64) NOT NULL,
    chunk_seq INT NOT NULL,
    content TEXT NOT NULL,
    page_num INT,
    INDEX idx_document (document_id),
    INDEX idx_vector (vector_id)
);

CREATE TABLE chat_session (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    tenant_id VARCHAR(64) NOT NULL,
    title VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_time (user_id, created_at)
);

CREATE TABLE chat_message (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    session_id BIGINT NOT NULL,
    role VARCHAR(16) NOT NULL,  -- user/assistant/system
    content MEDIUMTEXT NOT NULL,
    citations_json JSON,
    tokens INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_session_time (session_id, created_at)
);

CREATE TABLE usage_record (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT,
    tenant_id VARCHAR(64),
    model VARCHAR(64),
    prompt_tokens INT,
    completion_tokens INT,
    cost_yuan DECIMAL(10,6),
    request_id VARCHAR(64),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_tenant_day (tenant_id, created_at)
);
```

## 3. JWT 认证最简实现

```java
@Component
public class JwtUtil {
    @Value("${jwt.secret}") private String secret;
    @Value("${jwt.expiration:86400000}") private long expirationMs;

    public String generate(Long userId, String tenantId, String role) {
        return Jwts.builder()
                .setSubject(String.valueOf(userId))
                .claim("tenantId", tenantId)
                .claim("role", role)
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + expirationMs))
                .signWith(SignatureAlgorithm.HS256, secret.getBytes(StandardCharsets.UTF_8))
                .compact();
    }
    public Claims parse(String token) {
        return Jwts.parser().setSigningKey(secret.getBytes(StandardCharsets.UTF_8))
                .parseClaimsJws(token).getBody();
    }
}
```

> 💡 **不必从零写**：用 Spring Security 6 + jjwt + 一个 `JwtAuthenticationFilter` 过滤器。**网上有大量模板**——本阶段不必重复造轮子。
